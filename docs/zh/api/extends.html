# 扩展成员

本节内容介绍组件库针对.NET内置对象的扩展函数。阅读前，建议先阅读指南open in new window。

# DateTime

System.DateTime 扩展函数

public void Example()
{
    var datetime = DateTime.Now;
    //返回类型为long的时间戳
    var timestamp1 = datetime.ToUtcTimeStamp();
    //返回类型为long的时间戳（13位，精确到毫秒） TimeStampUnit:组件库内置枚举
    var timestamp2 = datetime.ToUtcTimeStamp(TimeStampUnit.MilliSecond);
}

# Enum

System.Enum 扩展函数

public void Example()
{
    //将Enum转为Int32
    var number = TimeStampUnit.Second.ToInt32();
    //获取Enum描述（System.ComponentModel.Description），没有描述会返回空字符串
    var description = TimeStampUnit.Second.GetDescription();

# String

System.String 扩展函数

public void Example()
{
    var value = "sugar.net";
    //将当字符串为null时，转为string.Empty
    var empty = value.ToEmptyIfNull();
    //将当字符串为string.Empty时，转为null
    var nullValue = value.ToNullIfEmpty();
    //判断字符是否为空
    var isnull = value.IsNullOrWhiteSpace();
    //判断字符串是否存在任何值
    var hasAnyValue = value.HasAnyValue();
    //替换首个字符
    var first = value.ReplaceFirst("s", "a");
    //替换结尾字符
    var last = value.ReplaceLast("t", "a");
}
